function DownloadDll {
    $payload = "AA"
    $outFile = "C:\temp\output.exe"
    
    # Ensure the directory exists
    if (!(Test-Path -Path "C:\temp\")) {
        New-Item -ItemType Directory -Path "C:\temp\"
    }
   
    # Convert the hexadecimal payload to a byte array
    $bytes = -split ($payload -replace '..', '0x$& ') | ForEach-Object { [System.Convert]::ToByte($_, 16) }

    # Write the byte array to the file
    [System.IO.File]::WriteAllBytes($outFile, $bytes)

    Write-output "DLL Download done...."

    # Open the DLL file with Notepad
    Start-Process -FilePath $outFile
}
